#define STATE_ON 0
#define STATE_OFF 1

int state = STATE_OFF;
int previousValue = 0;

void setup() {
  // put your setup code here, to run once:
  pinMode(3, INPUT_PULLUP);
  pinMode(12, OUTPUT);
}

void loop() {
  // put your main code here, to run repeatedly:
  int value = !digitalRead(3);
  if(value == 1 && previousValue == 0)
  {
    if(state == STATE_ON)
    {
      state = STATE_OFF;
    }else
    {
      state = STATE_ON;
    }
    delay(100);
  }
  previousValue = value;
  switch(state)
  {
    case STATE_ON:
    on();
    break;
    case STATE_OFF:
    off();
    break;
  }
}

void on()
{
  digitalWrite(12, HIGH);
}
void off()
{
  digitalWrite(12, LOW);
}
