#define STATE_ON 0
#define STATE_OFF 1

int state;
int value;

void setup() {
  // put your setup code here, to run once:
  pinMode(3, INPUT_PULLUP);
  pinMode(12, OUTPUT);
}

void loop() {
  // put your main code here, to run repeatedly:
  int value = !digitalRead(3);
  if(value == 1)
  {
    if(state == STATE_ON)
    {
      state = STATE_OFF;
    }else
    {
      state = STATE_ON;
    }
  }
  switch(state)
  {
    case STATE_ON:
    on();
    break;
    case STATE_OFF:
    off();
    break;
  }
}

void on()
{
  digitalWrite(12, HIGH);
}
void off()
{
  digitalWrite(12, LOW);
}
